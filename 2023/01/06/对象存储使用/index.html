дёҠдј жөҒзЁӢ

ж–Үд»¶дёҠдј еҲҶдёәе®ўжҲ·з«ҜдёҠдј пјҲдё»иҰҒжҳҜжҢҮзҪ‘йЎөз«Ҝе’Ңз§»еҠЁз«Ҝзӯүйқўеҗ‘з»Ҳз«Ҝз”ЁжҲ·зҡ„еңәжҷҜпјүе’ҢжңҚеҠЎз«ҜдёҠдј дёӨз§ҚеңәжҷҜпјҢе…·дҪ“еҸҜд»ҘеҸӮиҖғж–ҮжЎЈдёҡеҠЎжөҒзЁӢгҖӮ

жңҚеҠЎз«ҜSDKеңЁдёҠдј ж–№йқўдё»иҰҒжҸҗдҫӣдёӨз§ҚеҠҹиғҪпјҢдёҖз§ҚжҳҜз”ҹжҲҗе®ўжҲ·з«ҜдёҠдј жүҖйңҖиҰҒзҡ„дёҠдј еҮӯиҜҒпјҢеҸҰеӨ–дёҖз§ҚжҳҜзӣҙжҺҘдёҠдј ж–Үд»¶еҲ°дә‘з«ҜгҖӮ

е®ўжҲ·з«ҜдёҠдј еҮӯиҜҒ

е®ўжҲ·з«ҜпјҲз§»еҠЁз«ҜжҲ–иҖ…Webз«ҜпјүдёҠдј ж–Үд»¶зҡ„ж—¶еҖҷпјҢйңҖиҰҒд»Һе®ўжҲ·иҮӘе·ұзҡ„дёҡеҠЎжңҚеҠЎеҷЁиҺ·еҸ–дёҠдј еҮӯиҜҒпјҢиҖҢиҝҷдәӣдёҠдј еҮӯиҜҒжҳҜйҖҡиҝҮжңҚеҠЎз«Ҝзҡ„SDKжқҘз”ҹжҲҗзҡ„пјҢз„¶еҗҺйҖҡиҝҮе®ўжҲ·иҮӘе·ұзҡ„дёҡеҠЎAPIеҲҶеҸ‘з»ҷе®ўжҲ·з«ҜдҪҝз”ЁгҖӮж №жҚ®дёҠдј зҡ„дёҡеҠЎйңҖжұӮдёҚеҗҢпјҢдёғзүӣдә‘ Go SDKж”ҜжҢҒдё°еҜҢзҡ„дёҠдј еҮӯиҜҒз”ҹжҲҗж–№ејҸгҖӮ

жӢ·иҙқ
// еӯҳеӮЁзӣёе…іеҠҹиғҪзҡ„еј•е…ҘеҢ…еҸӘжңүиҝҷдёӨдёӘпјҢеҗҺйқўдёҚеҶҚиөҳиҝ°
import (
вҖңgithub.com/qiniu/go-sdk/v7/auth/qboxвҖқ
вҖңgithub.com/qiniu/go-sdk/v7/storageвҖқ
)

accessKey := вҖңyour access keyвҖқ
secretKey := вҖңyour secret keyвҖқ
mac := qbox.NewMac(accessKey, secretKey)

з®ҖеҚ•дёҠдј зҡ„еҮӯиҜҒ

жңҖз®ҖеҚ•зҡ„дёҠдј еҮӯиҜҒеҸӘйңҖиҰҒAccessKeyпјҢSecretKeyе’ҢBucketе°ұеҸҜд»ҘгҖӮ

жӢ·иҙқ
bucket:=вҖқyour bucket nameвҖқ
putPolicy := storage.PutPolicy{
Scope: bucket,
}
mac := qbox.NewMac(accessKey, secretKey)
upToken := putPolicy.UploadToken(mac)

й»ҳи®Өжғ…еҶөдёӢпјҢеңЁдёҚжҢҮе®ҡдёҠдј еҮӯиҜҒзҡ„жңүж•Ҳж—¶й—ҙжғ…еҶөдёӢпјҢй»ҳи®Өжңүж•Ҳжңҹдёә1дёӘе°Ҹж—¶гҖӮд№ҹеҸҜд»ҘиҮӘиЎҢжҢҮе®ҡдёҠдј еҮӯиҜҒзҡ„жңүж•ҲжңҹпјҢдҫӢеҰӮпјҡ

жӢ·иҙқ
//иҮӘе®ҡд№үеҮӯиҜҒжңүж•ҲжңҹпјҲзӨәдҫӢ2е°Ҹж—¶пјҢExpires еҚ•дҪҚдёәз§’пјҢдёәдёҠдј еҮӯиҜҒзҡ„жңүж•Ҳж—¶й—ҙпјү
bucket := вҖңyour bucket nameвҖқ

putPolicy := storage.PutPolicy{
Scope: bucket,
}
putPolicy.Expires = 7200 //зӨәдҫӢ2е°Ҹж—¶жңүж•Ҳжңҹ
mac := qbox.NewMac(accessKey, secretKey)
upToken := putPolicy.UploadToken(mac)

иҰҶзӣ–дёҠдј зҡ„еҮӯиҜҒ

иҰҶзӣ–дёҠдј йҷӨдәҶйңҖиҰҒз®ҖеҚ•дёҠдј жүҖйңҖиҰҒзҡ„дҝЎжҒҜд№ӢеӨ–пјҢиҝҳйңҖиҰҒжғіиҝӣиЎҢиҰҶзӣ–зҡ„ж–Үд»¶еҗҚз§°пјҢиҝҷдёӘж–Үд»¶еҗҚз§°еҗҢж—¶еҸҜжҳҜе®ўжҲ·з«ҜдёҠдј д»Јз ҒдёӯжҢҮе®ҡзҡ„ж–Үд»¶еҗҚпјҢдёӨиҖ…еҝ…йЎ»дёҖиҮҙгҖӮ

жӢ·иҙқ
bucket := вҖңyour bucket nameвҖқ

// йңҖиҰҒиҰҶзӣ–зҡ„ж–Үд»¶еҗҚ
keyToOverwrite := вҖңqiniu.mp4вҖқ
putPolicy := storage.PutPolicy{
Scope: fmt.Sprintf(вҖң%s:%sвҖқ, bucket, keyToOverwrite),
}
mac := qbox.NewMac(accessKey, secretKey)
upToken := putPolicy.UploadToken(mac)

иҮӘе®ҡд№үдёҠдј еӣһеӨҚзҡ„еҮӯиҜҒ

й»ҳи®Өжғ…еҶөдёӢпјҢж–Үд»¶дёҠдј еҲ°дёғзүӣд№ӢеҗҺпјҢеңЁжІЎжңүи®ҫзҪ®returnBodyжҲ–иҖ…еӣһи°ғзӣёе…ізҡ„еҸӮж•°жғ…еҶөдёӢпјҢдёғзүӣиҝ”еӣһз»ҷдёҠдј з«Ҝзҡ„еӣһеӨҚж јејҸдёәhashе’ҢkeyпјҢдҫӢеҰӮпјҡ

жӢ·иҙқ
{вҖңhashвҖқ:вҖқFtgm-CkWePC9fzMBTRNmPMhGBcSVвҖқ,вҖқkeyвҖқ:вҖқqiniu.jpgвҖқ}

жңүж—¶еҖҷжҲ‘д»¬еёҢжңӣиғҪиҮӘе®ҡд№үиҝҷдёӘиҝ”еӣһзҡ„JSONж јејҸзҡ„еҶ…е®№пјҢеҸҜд»ҘйҖҡиҝҮи®ҫзҪ®returnBodyеҸӮж•°жқҘе®һзҺ°пјҢеңЁreturnBodyдёӯпјҢжҲ‘д»¬еҸҜд»ҘдҪҝз”Ёдёғзүӣж”ҜжҢҒзҡ„йӯ”жі•еҸҳйҮҸе’ҢиҮӘе®ҡд№үеҸҳйҮҸгҖӮ

жӢ·иҙқ
bucket := вҖңyour bucket nameвҖқ

putPolicy := storage.PutPolicy{
Scope: bucket,
ReturnBody: {"key":"$(key)","hash":"$(etag)","fsize":$(fsize),"bucket":"$(bucket)","name":"$(x:name)"},
}
mac := qbox.NewMac(accessKey, secretKey)
upToken := putPolicy.UploadToken(mac)

еҲҷж–Үд»¶дёҠдј еҲ°дёғзүӣд№ӢеҗҺпјҢж”¶еҲ°зҡ„еӣһеӨҚеҶ…е®№ж јејҸеҰӮдёӢпјҡ

жӢ·иҙқ
{вҖңkeyвҖқ:вҖқgithub-x.pngвҖқ,вҖқhashвҖқ:вҖқFqKXVdTvIx_mPjOYdjDyUSy_H1jrвҖқ,вҖқfsizeвҖқ:6091,вҖқbucketвҖқ:вҖқif-pblвҖқ,вҖқnameвҖқ:вҖқgithub logoвҖқ}

еҜ№дәҺдёҠйқўзҡ„иҮӘе®ҡд№үиҝ”еӣһеҖјпјҢжҲ‘д»¬йңҖиҰҒиҮӘе®ҡд№үз»“жһ„дҪ“жқҘи§ЈжһҗиҝҷдёӘеӣһеӨҚпјҢдҫӢеҰӮдёӢйқўжҸҗдҫӣдәҶдёҖдёӘи§Јжһҗз»“жһңзҡ„ж–№жі•пјҡ

жӢ·иҙқ
// иҮӘе®ҡд№үиҝ”еӣһеҖјз»“жһ„дҪ“
type MyPutRet struct {
Key string
Hash string
Fsize int
Bucket string
Name string
}

localFile := вҖңyour local file pathвҖқ
bucket := вҖңyour bucket nameвҖқ
key := вҖңyour file save keyвҖқ

// дҪҝз”Ё returnBody иҮӘе®ҡд№үеӣһеӨҚж јејҸ
putPolicy := storage.PutPolicy{
Scope: bucket,
ReturnBody: {"key":"$(key)","hash":"$(etag)","fsize":$(fsize),"bucket":"$(bucket)","name":"$(x:name)"},
}
mac := qbox.NewMac(accessKey, secretKey)
upToken := putPolicy.UploadToken(mac)

cfg := storage.Config{}
formUploader := storage.NewFormUploader(&cfg)
ret := MyPutRet{}
putExtra := storage.PutExtra{
Params: map[string]string{
вҖңx:nameвҖқ: вҖңgithub logoвҖқ,
},
}
err := formUploader.PutFile(context.Background(), &ret, upToken, key, localFile, &putExtra)
if err != nil {
fmt.Println(err)
return
}
fmt.Println(ret.Bucket, ret.Key, ret.Fsize, ret.Hash, ret.Name)
